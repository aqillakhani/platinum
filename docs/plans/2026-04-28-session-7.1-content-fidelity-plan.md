```python
from fastapi import UploadFile, File, Form

@app.post("/clip-sim")
async def clip_sim(
    image: UploadFile = File(...),
    prompt: str = Form(...),
    scorer = Depends(get_scorer),
) -> dict:
    if not prompt:
        return JSONResponse({"error": "prompt required"}, status_code=400)
    image_bytes = await image.read()
    if not image_bytes:
        return JSONResponse({"error": "image bytes required"}, status_code=400)
    sim = scorer.clip_similarity(image_bytes=image_bytes, prompt=prompt)
    return {"similarity": float(sim)}
```

In model.py:

```python
def encode_image_text_similarity(image_bytes: bytes, prompt: str) -> float:
    """Cosine similarity of CLIP image and text embeddings."""
    img = preprocess(load_image_bytes(image_bytes)).unsqueeze(0).to(device)
    with torch.no_grad():
        img_feat = clip_model.encode_image(img)
        tokens = clip_tokenizer([prompt]).to(device)
        txt_feat = clip_model.encode_text(tokens)
        img_feat = img_feat / img_feat.norm(dim=-1, keepdim=True)
        txt_feat = txt_feat / txt_feat.norm(dim=-1, keepdim=True)
        sim = (img_feat @ txt_feat.T).item()
    return float(sim)
```

(The Scorer class wraps this; Fake scorer returns deterministic values.)

```python
    "input_schema": {
        "type": "object",
        "required": ["score", "missing", "rationale"],
        "properties": {

class ClaudeContentChecker:
    def __init__(self, *, model: str = "claude-haiku-4-5",
```

```python
def extract_character_names(scenes: list[Scene]) -> list[str]:
    """Heuristic: collect proper-noun phrases that appear in 2+ scenes."""
    candidates: dict[str, int] = {}
    for s in scenes:
        text = s.narration_text
        # Find capitalized words that aren't sentence-initial
        for match in re.finditer(r'(?<!^)(?<![.!?] )([A-Z][a-z]+(?: [A-Z][a-z]+)*)', text):
            name = match.group(0)
            candidates[name] = candidates.get(name, 0) + 1
    # Keep names appearing in 2+ scenes
    return [name for name, count in candidates.items() if count >= 2]
```

Test:
```python

    scenes = [
        Scene(index=1, narration_text="Fortunato laughed at Montresor."),
        Scene(index=2, narration_text="Montresor smiled. Fortunato did not."),
    ]
    assert set(extract_character_names(scenes)) == {"Fortunato", "Montresor"}

def test_extract_character_names_ignores_one_off_capitalized_words():
```
